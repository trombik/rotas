def initialize(arg)
  @name = arg[:name]
  @path = arg[:path].is_a?(Pathname) ? arg[:path] : Pathname.new(arg[:path])
  @stable_version = arg[:stable_version]
  @arch = {}
  arg[:arch].each do |a|
    @arch[a.to_sym] = Rotas::Project::Arch.new(
      arch: a, path: @path.realpath + a, project: @name,
      stable_version: @stable_version
    )
  end
end
